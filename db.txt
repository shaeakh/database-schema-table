
CREATE TABLE tbl_post(
	Post_id int AUTO_INCREMENT PRIMARY KEY,
    Event_id int,
    Ptime time,
    details varchar(100),    
    Venue_id int,
    FOREIGN KEY (Event_id) REFERENCES tbl_event(Event_id),
    FOREIGN KEY (Venue_id) REFERENCES tbl_venue(Venue_id)
)



CREATE TABLE tbl_expense_diary(
	transaction_id int PRIMARY KEY,
    Event_id int,
    Etime time,
    Donate_id int,
    Receipt_id int,
    Treasurer_id int,
    debited int,
    credited int,
    current_balance int,  
    FOREIGN KEY (Donate_id) REFERENCES tbl_donate(Donate_id),
    FOREIGN KEY (Event_id) REFERENCES tbl_event(Event_id),
    FOREIGN KEY (Receipt_id) REFERENCES tbl_receipt(Receipt_id),
    FOREIGN KEY (Treasurer_id) REFERENCES tbl_treasurer(Treasurer_id)
    
)

CREATE TABLE tbl_event(
	Event_id int AUTO_INCREMENT PRIMARY KEY,
    Event_type varchar(100),
	Edate date,
    Start_time time,
    End_time time,
    Details varchar(100),
    Application_id int,
    Venue_id int,
    FOREIGN KEY (Application_id) REFERENCES tbl_application(Application_id),
    FOREIGN KEY (Venue_id) REFERENCES tbl_venue(Venue_id)
)

CREATE TABLE tbl_application(
	Application_id int AUTO_INCREMENT PRIMARY KEY,
    Reason varchar(200),
    Atime time,
   Proctor_Name varchar(100),
    Proctor_Phone_number varchar(100),
    Proctor_Email varchar(100)    
)

CREATE TABLE tbl_invitation(
	Invite_id int AUTO_INCREMENT PRIMARY KEY,
    ITime time,
    IDate date,
    Event_name varchar(100),
    details varchar(100),
    Venue_id int,
    FOREIGN KEY (Venue_id) REFERENCES tbl_venue(Venue_id)
    
)

CREATE TABLE tbl_Venue_booking(
	Booking_id int PRIMARY KEY,
    Venue_id int,
    Btime time,
    Start_time time,
    End_time time, 
    FOREIGN KEY (Venue_id) REFERENCES tbl_venue(Venue_id)
)
CREATE TABLE tbl_venue(
	Venue_id int AUTO_INCREMENT PRIMARY KEY,
    Name varchar(100), 
    Location varchar(100)
)

CREATE TABLE tbl_organiser(
	organiser_id int AUTO_INCREMENT PRIMARY KEY,
    Name varchar(100),
    Phone_number varchar(100),  
    Email  varchar(100),  
    Department varchar(100)
)


CREATE TABLE tbl_treasurer(
	Treasurer_id int AUTO_INCREMENT PRIMARY KEY,
    Name varchar(100),
    Phone_number varchar(100),  
    Email  varchar(100),  
    Department varchar(100)
)

CREATE TABLE tbl_donate(
	Donate_id int AUTO_INCREMENT PRIMARY KEY,
    Member_ID int,
    Amount int,
    DTime time,
    Reference  varchar(100),

    FOREIGN KEY (Member_ID) REFERENCES tbl_memberlist(Member_ID)
)

CREATE TABLE tbl_memberlist(
	Member_ID int AUTO_INCREMENT PRIMARY KEY,
    Member_Type varchar(100),
    Name varchar(100),
    Phone_number varchar(100),  
    Email  varchar(100),  
    Department varchar(100),    
    Registration_Number int,
    Blood_group varchar(20),            
    Location varchar(100)
)

CREATE TABLE tbl_receipt(
	Receipt_id int PRIMARY KEY, 
    Service_id int,
    Item_id int,
    rTime time,
    cost int,
    details varchar(100),
    
    FOREIGN KEY (Service_id) REFERENCES tbl_service(Service_id),
    FOREIGN KEY (Item_id) REFERENCES tbl_item(Item_id)
)

CREATE TABLE tbl_item(
	Item_id int AUTO_INCREMENT PRIMARY KEY, 
    Item_name varchar(40), 
    cost int,
    Service_id int ,
    FOREIGN KEY (Service_id) REFERENCES tbl_service(Service_id)
)


CREATE TABLE tbl_service(
	Service_id int PRIMARY KEY, 
    Name varchar(100),
    Location varchar(100),
    fees int
)
